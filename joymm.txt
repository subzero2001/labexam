addition of two 16bit numbers


DATA SEGMENT
           OPR1 DW 1234H
           OPR2 DW 5678H
            RESULT DW ?
DATA ENDS

CODE SEGMENT

START: MOV AX, DATA
            MOV DS,AX
            MOV AX,OPR1
            MOV BX,OPR2
            ADD AX,BX
            MOV RESULT,AX
            MOV AH,4CH
            INT 21H
CODE ENDS
END START



string palindrome


DATA SEGMENT
	MYSTRING DB 'MALAYALAM$'
	STRING1 DB 'PALINDROME$'
	STRING2 DB 'NOT PALINDROME$'
DATA ENDS
CODE SEGMENT
ASSUME CS:CODE,DS:DATA

	START: MOV AX,DATA
		MOV DS,AX
		LEA DI,MYSTRING
		MOV SI,DI
	BACK: MOV AL,[DI]
		CMP AL,'$'
		JZ BACK1
		INC DI
		JMP BACK
	BACK1: DEC DI
		CMP SI,DI
             

		MOV AL, [SI]
		CMP AL,[DI]
		JNZ PRINT2
		INC SI
		JMP BACK1
	PRINT1: LEA DX, STRING1
		JMP L2
	PRINT2: LEA DX,STRING2
	    L2: MOV AH,09H
		INT 21H
		MOV AH,4CH
		INT 21H
CODE ENDS
END START
